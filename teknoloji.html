// A* pathfinding — risk ağırlıklı maliyet fonksiyonu
// Diagonal hareket √2 maliyetli, risk >0.9 → duvar
function computeRoute(start, end, riskMatrix) {
  const openSet = [start];
  const cameFrom = {}, gScore = {}, fScore = {};

  // Maliyet hesabı: moveCost * (1 + risk * 10)
  // risk skoru U-Net segmentasyon çıktısından
  for (const neighbor of getNeighbors(current)) {
    const risk = riskMatrix[neighbor.r][neighbor.c];
    if (risk > 0.9) continue; // geçilemez

    const moveCost = isDiagonal ? 1.414 : 1;
    const tentativeG = gScore[ck] + moveCost * (1 + risk * 10);

    if (tentativeG < gScore[nk]) {
      cameFrom[nk] = current;
      gScore[nk] = tentativeG;
      fScore[nk] = tentativeG + manhattan(neighbor, end);
    }
  }
  return reconstructPath(cameFrom, end);
}
